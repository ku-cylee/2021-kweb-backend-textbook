\usepackage{environ}
\usepackage{ifthen}

\newcommand{\intand}[2]{\ifnum #1=0 0\else #2 \fi}
\newcommand{\intor}[2]{\ifnum #1=1 1\else #2 \fi}

\ifthenelse{\isfrontend=1}{
    \ifthenelse{\isbackend=1}{
        \newcommand{\recipecond}[3]{#3}
    }{
        \newcommand{\recipecond}[3]{#1}
    }
}{
    \ifthenelse{\isbackend=1}{
        \newcommand{\recipecond}[3]{#2}
    }{
        \newcommand{\recipecond}[3]{0}
    }
}

\newcommand\recipeoptioncmd[4]{\ifthenelse{\recipecond{#1}{#2}{#3}=1}{#4}{}}

\NewEnviron{iffront}{\recipeoptioncmd{1}{0}{0}{\BODY}}
\NewEnviron{ifback}{\recipeoptioncmd{0}{1}{0}{\BODY}}
\NewEnviron{ifboth}{\recipeoptioncmd{0}{0}{1}{\BODY}}
\NewEnviron{ifnotfront}{\recipeoptioncmd{0}{1}{1}{\BODY}}
\NewEnviron{ifnotback}{\recipeoptioncmd{1}{0}{1}{\BODY}}
\NewEnviron{ifnotboth}{\recipeoptioncmd{1}{1}{0}{\BODY}}
